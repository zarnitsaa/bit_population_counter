	end

	begin
			begin
			for (i=0; i<=CNT; i++)
			wait (!data_val_o)
			@(posedge clk_i_ext)

	begin
	@(posedge clk_i_ext)
			begin
			for (i=0; i<=CNT; i++)
				begin
				assert ($countbits(rand_queue[i],'1)==output_queue[i]
